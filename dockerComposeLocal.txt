version: '3.4'
services:
  db:
    image: gustavotudeia/mysql-spring-boot-with-docker
    command: mysqld --default-authentication-plugin=mysql_native_password
    restart: always
    build:
      context: .
      dockerfile: Dockerfile
    environment:
      TZ: America/Sao_Paulo
      MYSQL_ROOT_PASSWORD: docker
      MYSQL_USER: docker
      MYSQL_PASSWORD: docker
      MYSQL_DATABASE: spring_boot_with_docker
    ports:
      - "3306:3308"
    networks:
      - spring-docker-network
  spring-boot-with-docker:
    image: gustavotudeia/spring-boot-with-docker
    restart: always
    build: ./spring-boot-with-docker
    working_dir: /spring-boot-with-docker
    environment:
      TZ: America/Sao_Paulo
      SPRING_BOOT_ENVIRONMENT: Production
    volumes:
      - ./spring-boot-with-docker:/spring-boot-with-docker
      - ~/.m2:/root/.m2
    ports:
      - "8080:8080"
    command: mvn clean spring-boot:run
    links:
      - db
    depends_on:
      - db
    networks:
      - spring-docker-network
networks:
  spring-docker-network:
    driver: bridge
